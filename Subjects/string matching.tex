\section{String Matching}
\subsection{Finite Automata}
This is a way to match strings (similar to FSM). When building the graph, just
remember that each nodes needs a edge pointing out for each letter in the
alphabet.

\textbf{Example:} Finite Automaton for pattern "abab". \emph{Note:} if a character
path you want to take isn't shown, go back to the beginning.

\begin{center}
\begin{tikzpicture}[
    node distance=2cm,
    state/.style={circle, draw, minimum size=1cm},
    accepting/.style={circle, draw, double, minimum size=1cm}
]
    % States
    \node[state] (q0) {0};
    \node[state, right of=q0] (q1) {1};
    \node[state, right of=q1] (q2) {2};
    \node[state, right of=q2] (q3) {3};
    \node[accepting, right of=q3] (q4) {4};

    % Main transitions (pattern matching)
    \draw[-{Stealth}] (q0) -- node[above] {$a$} (q1);
    \draw[-{Stealth}] (q1) -- node[above] {$b$} (q2);
    \draw[-{Stealth}] (q2) -- node[above] {$a$} (q3);
    \draw[-{Stealth}] (q3) -- node[above] {$b$} (q4);

    % Failure transitions
    \draw[-{Stealth}] (q0) edge[loop above] node[above] {$b$} (q0);
    \draw[-{Stealth}] (q1) edge[loop above] node[above] {$a$} (q1);
    \draw[-{Stealth}] (q2) edge[bend left=60] node[below] {$b$} (q0);
    \draw[-{Stealth}] (q3) edge[bend left=70] node[below] {$a$} (q1);
    \draw[-{Stealth}] (q4) edge[bend left=40] node[above] {$a$} (q1);
    \draw[-{Stealth}] (q4) edge[bend right=80] node[below] {$b$} (q2);
\end{tikzpicture}
\end{center}

\subsection{Knuth–Morris–Pratt (KMP)}
Given a prefix $\pi$ for a string with length $i$, then build a graph from that pattern.

\textbf{Example:} KMP for pattern ``abab''

\begin{figure}
    \begin{table}[H]
        \centering
        \begin{tabular}{c|c|c|c|c}
            $i$ & 1 & 2 & 3 & 4\\
            \hline
            $\pi$[$i$] & 0 & 0 & 1 & 2\\
             \hline
            \verb|P[i]| & a & b & a & b \\
        \end{tabular}
    \end{table}
    \caption{Prefix function table}
\end{figure}

\begin{figure}
    
    \begin{center}
    \begin{tikzpicture}[
        node distance=2.5cm,
        state/.style={circle, draw, minimum size=1cm},
        accepting/.style={circle, draw, double, minimum size=1cm}
    ]
        % States
        \node[state] (q0) {0};
        \node[state, right of=q0] (q1) {1};
        \node[state, right of=q1] (q2) {2};
        \node[state, right of=q2] (q3) {3};
        \node[accepting, right of=q3] (q4) {4};

        % Main transitions (matching)
        \draw[-{Stealth}, thick] (q0) -- node[above] {$a$} (q1);
        \draw[-{Stealth}, thick] (q1) -- node[above] {$b$} (q2);
        \draw[-{Stealth}, thick] (q2) -- node[above] {$a$} (q3);
        \draw[-{Stealth}, thick] (q3) -- node[above] {$b$} (q4);

        % Failure links (dashed)
        \draw[-{Stealth}, dashed, red] (q2) edge[bend right=50] node[below] {$\pi[2]=0$} (q0);
        \draw[-{Stealth}, dashed, red] (q3) edge[bend right=40] node[below] {$\pi[3]=1$} (q1);
        \draw[-{Stealth}, dashed, red] (q4) edge[bend left=50] node[above] {$\pi[4]=2$} (q2);
    \end{tikzpicture}
    \end{center}
    \caption{Automaton for KMP}
\end{figure}

The solid arrows show character transitions, while dashed red arrows show failure links ($\pi$ values).

\subsection{String Matching Overview}
\begin{table}[H]
    \centering
    \small
    \begin{tabular}{l|c|c|c|p{5cm}}
        & Construction Time & Matching Time & Total Time & Explanation \\
        \hline
        Finite Automaton &  $\BigO(m\cdot|\Sigma|)$ & $\BigO(n)$ & $\BigO(m\cdot|\Sigma|+n)$  & $\Sigma$ is the size of the alphabet, $m$ is the size of the pattern\\
        \hline
        KMP & $\BigO(m)$ & $\BigO(n)$ & $\BigO(n+m)$ & $n$ is the size of the string, $m$ is the size of the pattern \\
    \end{tabular}
\end{table}
